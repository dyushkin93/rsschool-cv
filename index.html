double sort(double **a, const int n, const int m)
{
  int c, i, j, x;
  for(i=0; i< n; i++)
    for(x=1; x< m; x++)
      for(j=0; j< m-x; j++)
        if (a[i][j] > a[i][j+1])
        {
          c = a[i][j];
          a[i][j] = a[i][j+1];
          a[i][j+1] = c;
        }
  return **a;
}
